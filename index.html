js-img-magnify

Pure JavaScript utility enabling magnifying glass effect on an images. No external dependencies required.

example1



      magnifyImg({
        src: '/assets/waterDrop.jpg',
        zoom: 3,
        target: document.getElementById('example1'),
        width: 500,
        MagnifyDomWidth: 200,
      })

example2



      magnifyImg({
        src: '/assets/waterDrop.jpg',
        zoom: 3,
        target: document.getElementById('example2'),
        width: 500,
        MagnifyDomWidth: 200,
        MagnifyDomStyles: {
          borderRadius: '50%',
          border: '4px solid red'
        }
      })
